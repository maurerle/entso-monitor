def revReplaceStr(string):
    '''
    replaces series names for better visualization
    '''
    st = str.replace(string, 'sum(', '')
    st = str.replace(st, ')', '')
    st = str.replace(st, '(', '')
    st = str.replace(st, '_', ' ')
    st = st.strip()

    return st
